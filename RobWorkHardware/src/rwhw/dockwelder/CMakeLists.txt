
OPTION(COMPONENT_DOCKWELDER_ENABLE "Enable DockWelder component" ${COMPONENT_DOCKWELDER_ENABLE})
IF( COMPONENT_DOCKWELDER_ENABLE )
    MESSAGE(STATUS "RobWorkHardware: DockWelder component ENABLED")    
    INCLUDE_DIRECTORIES(${Boost_ASIO_INCLUDE_DIR}) 
    ADD_LIBRARY(rwhw_dockwelder DockWelder.cpp DockWelder.hpp)
    INSTALL(TARGETS rwhw_dockwelder DESTINATION ${RWHW_LIB_INSTALL_DIR})
    set( ENV{ROBWORKHARDWARE_LIBRARIES} "$ENV{ROBWORKHARDWARE_LIBRARIES}rwhw_dockwelder;")
ELSE()
    MESSAGE(STATUS "RobWorkHardware: DockWelder component DISABLED")
ENDIF()
